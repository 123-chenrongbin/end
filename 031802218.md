;分别使用计数器T0和T1产生周期为1ms的连续方波和负脉冲，
;并在示波器上观察

DATA SEGMENT
T0 EQU 280H
T1 EQU 281H
CEL_53 EQU 283H
DATA ENDS
CODE SEGMENT
  ASSUME CS:CODE,DS:DATA

START:
  MOV DX,CEL_53    ;计数器0，产生1ms的连续方波
  MOV AL,00110110B  
  OUT DX,AL
  MOV DX,T0
  MOV AX,1000
  OUT DX,AL
  MOV AL,AH
  OUT DX,AL
  
  MOV DX,CEL_53   ;计数器1，产生1ms的负脉冲
  MOV AL,01110110B
  OUT DX,AL
  MOV DX,T1
  MOV AX,1000
  OUT DX,AL
  MOV AL,AH
  OUT DX,AL

  MOV AH,4CH
  INT 21H

  CODE ENDS
  END START
